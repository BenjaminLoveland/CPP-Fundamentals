\begin{itemize}

\section{Enumerations}\label{enumerations}}

\begin{itemize}
\tightlist
\item
  https://docs.microsoft.com/en-us/cpp/cpp/enumerations-cpp?view=msvc-160
\item
  an enumeration is a user-defined type that consists of a set of named
  integral constants that are known as enumerators
\end{itemize}

\hypertarget{defining-enumeration-types}{%
\subsection{Defining enumeration
types}\label{defining-enumeration-types}}

\begin{itemize}
\tightlist
\item
  syntax to declare enumeration type:
\end{itemize}

\begin{Shaded}
\begin{Highlighting}[]
\KeywordTok{enum}\NormalTok{ Name }\OperatorTok{\{}\NormalTok{name1}\OperatorTok{,}\NormalTok{ name2}\OperatorTok{,}\NormalTok{ name3}\OperatorTok{,} \OperatorTok{...\};}
\end{Highlighting}
\end{Shaded}

\begin{itemize}
\tightlist
\item
  each name in an enum type is assigned an integral value that
  corresponds to its place in the order of the values listed
\item
  by default, the first value is assigned \(0\), the next one is \(1\),
  and so on.
\item
  however, you can explictly set the value of an enumerator
\end{itemize}
